  <script>
    const API_URL = 'https://script.google.com/macros/s/AKfycbxBKWdrVL9antbcBEwZiehsL2PxwtKo3h09Rtn9Fz3qc0MVEWOmPkpxxN7f5XXqIDg/exec';

    const particlesContainer = document.getElementById('particles');
    for (let i = 0; i < 30; i++) {
      const p = document.createElement('div');
      p.className = 'particle';
      p.style.cssText = `
        left: ${Math.random() * 100}%;
        width: ${2 + Math.random() * 4}px;
        height: ${2 + Math.random() * 4}px;
        animation-duration: ${8 + Math.random() * 15}s;
        animation-delay: ${Math.random() * 12}s;
        opacity: ${0.3 + Math.random() * 0.5};
      `;
      particlesContainer.appendChild(p);
    }

    let selectedSlot = null;
    let currentSlots = [];

    function getSelectedPreference() {
      const checked = document.querySelector('input[name="gender"]:checked');
      return checked ? checked.value : '';
    }

    async function loadSlots() {
      const preference = getSelectedPreference();

      if (!preference) {
        currentSlots = [];
        renderSlots();
        document.getElementById('slotBadge').textContent = 'בחרו מטפל/ת קודם';
        return;
      }

      document.getElementById('slotBadge').textContent = 'טוען שעות...';

      try {
        const response = await fetch(`${API_URL}?action=slots&preference=${encodeURIComponent(preference)}`);
        const result = await response.json();

        if (!result.success) {
          throw new Error(result.error || 'שגיאה בטעינת סלוטים');
        }

        handleSlotsLoaded(result);
      } catch (error) {
        handleServerError(error);
      }
    }

    function handleSlotsLoaded(result) {
      currentSlots = (result.slots || []).filter(slot => slot.availableForPreference > 0);
      renderSlots();

      const totalLeft = result.totalLeft || 0;
      const totalCapacity = result.totalCapacity || 48;
      const pct = totalCapacity > 0 ? (totalLeft / totalCapacity) * 100 : 0;

      document.getElementById('spotsFill').style.width = pct + '%';
      document.getElementById('spotsLeft').textContent = totalLeft;
      document.getElementById('spotsBarText').textContent =
        `${totalLeft} מקומות פנויים מתוך ${totalCapacity}`;

      if (selectedSlot) {
        const selectedSlotObj = currentSlots.find(s => s.time === selectedSlot);
        if (!selectedSlotObj) {
          selectedSlot = null;
          document.getElementById('slotBadge').textContent = 'בחר שעה';
        }
      } else {
        document.getElementById('slotBadge').textContent = currentSlots.length ? 'בחר שעה' : 'אין שעות פנויות';
      }
    }

    function renderSlots() {
      const grid = document.getElementById('slotsGrid');
      grid.innerHTML = '';

      if (currentSlots.length === 0) {
        grid.innerHTML = '<div style="grid-column:1/-1;text-align:center;color:#6699bb;padding:20px;font-weight:700;">בחרו קודם מטפל או מטפלת כדי לראות שעות פנויות</div>';
        return;
      }

      currentSlots.forEach(slot => {
        const btn = document.createElement('button');
        btn.type = 'button';
        btn.className = 'slot-btn' + (selectedSlot === slot.time ? ' selected' : '');
        btn.innerHTML = `
          <span class="slot-time">${slot.time}</span>
          <span class="slot-avail">${slot.availableForPreference} פנוי</span>
        `;
        btn.onclick = () => selectSlot(slot.time);
        grid.appendChild(btn);
      });
    }

    function selectSlot(time) {
      selectedSlot = time;
      const slot = currentSlots.find(s => s.time === time);
      const avail = slot ? slot.availableForPreference : 0;

      document.getElementById('slotBadge').textContent = `${time} — ${avail} מקומות פנויים`;
      document.getElementById('fieldSlot').classList.remove('field-error');
      document.getElementById('slotError').style.display = 'none';

      renderSlots();
    }

    function validate() {
      let valid = true;

      const name = document.getElementById('fullName').value.trim();
      const nameField = document.getElementById('fieldName');
      if (name.length < 2) {
        nameField.classList.add('field-error');
        valid = false;
      } else {
        nameField.classList.remove('field-error');
      }

      const phone = document.getElementById('phone').value.trim().replace(/[^\d]/g, '');
      const phoneField = document.getElementById('fieldPhone');
      if (phone.length < 9) {
        phoneField.classList.add('field-error');
        valid = false;
      } else {
        phoneField.classList.remove('field-error');
      }

      const gender = document.querySelector('input[name="gender"]:checked');
      const genderField = document.getElementById('fieldGender');
      if (!gender) {
        genderField.classList.add('field-error');
        document.getElementById('genderError').style.display = 'block';
        valid = false;
      } else {
        genderField.classList.remove('field-error');
        document.getElementById('genderError').style.display = 'none';
      }

      const slotField = document.getElementById('fieldSlot');
      if (!selectedSlot) {
        slotField.classList.add('field-error');
        document.getElementById('slotError').style.display = 'block';
        valid = false;
      } else {
        slotField.classList.remove('field-error');
        document.getElementById('slotError').style.display = 'none';
      }

      return valid;
    }

    async function submitForm() {
      if (!validate()) return;

      const submitBtn = document.getElementById('submitBtn');
      submitBtn.disabled = true;
      submitBtn.textContent = 'שולח...';

      const payload = {
        action: 'submit',
        fullName: document.getElementById('fullName').value.trim(),
        phone: document.getElementById('phone').value.trim(),
        preference: document.querySelector('input[name="gender"]:checked').value,
        time: selectedSlot
      };

      try {
        const response = await fetch(API_URL, {
          method: 'POST',
          headers: {
            'Content-Type': 'text/plain;charset=utf-8'
          },
          body: JSON.stringify(payload)
        });

        const result = await response.json();
        handleSubmitSuccess(result);
      } catch (error) {
        handleSubmitFailure(error);
      }
    }

    function handleSubmitSuccess(response) {
      const submitBtn = document.getElementById('submitBtn');
      submitBtn.disabled = false;
      submitBtn.innerHTML = '🌿 אני רוצה עיסוי!';

      if (!response.success) {
        alert(response.message || response.error || 'אירעה שגיאה. נסו שוב.');
        loadSlots();
        return;
      }

      document.getElementById('formCard').style.display = 'none';
      const sc = document.getElementById('successCard');
      sc.style.display = 'block';

      const prefLabel = {
        male: 'מטפל',
        female: 'מטפלת'
      }[response.preference] || '';

      document.getElementById('successDetail').innerHTML = `
        <strong>${response.fullName}</strong>, נרשמתם בהצלחה!<br/>
        🕐 שעת עיסוי: <strong>${response.time}</strong><br/>
        👤 בחירה: <strong>${prefLabel}</strong><br/>
        <br/>נתראה בעמדה — תצאו רגועים ומחויכים 🌿
      `;

      loadSlots();
    }

    function handleSubmitFailure(error) {
      const submitBtn = document.getElementById('submitBtn');
      submitBtn.disabled = false;
      submitBtn.innerHTML = '🌿 אני רוצה עיסוי!';
      alert('שגיאה בשליחה: ' + (error.message || error));
    }

    function handleServerError(error) {
      console.error(error);
      document.getElementById('slotBadge').textContent = 'שגיאה בטעינת שעות';
      alert('שגיאה בטעינת השעות: ' + (error.message || error));
    }

    function resetForm() {
      document.getElementById('fullName').value = '';
      document.getElementById('phone').value = '';
      document.querySelectorAll('input[name="gender"]').forEach(r => r.checked = false);
      selectedSlot = null;

      ['fieldName','fieldPhone','fieldGender','fieldSlot'].forEach(id => {
        document.getElementById(id).classList.remove('field-error');
      });

      document.getElementById('formCard').style.display = 'block';
      document.getElementById('successCard').style.display = 'none';
      document.getElementById('slotBadge').textContent = 'בחרו מטפל/ת קודם';
      currentSlots = [];
      renderSlots();
      window.scrollTo({ top: 0, behavior: 'smooth' });
    }

    document.querySelectorAll('input[name="gender"]').forEach(radio => {
      radio.addEventListener('change', () => {
        selectedSlot = null;
        document.getElementById('slotBadge').textContent = 'טוען שעות...';
        loadSlots();
      });
    });

    document.getElementById('submitBtn').addEventListener('click', submitForm);

    renderSlots();
